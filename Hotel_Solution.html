import datetime
weekday_D = {0: 'Monday',1:'Tuesday',2:'Wednesday',3:'Thursday',4:'Friday',5:'Saturday',6:'Sunday'}

def get_weekday(reservation_date):
    
    week_index = datetime.datetime.strptime(reservation_date, "%m/%d/%Y").weekday()
    week_day = weekday_D[week_index]
    return week_day

odt["weekday"] = odt["reservation date"].apply(get_weekday)
